### 典型 AI 应用微服务拆分
```
                        ┌──────────────┐
                        │  API Gateway  │ (Kong / Nginx / Envoy)
                        └──────┬───────┘
               ┌───────────────┼───────────────┐
               ↓               ↓               ↓
        ┌────────────┐  ┌────────────┐  ┌────────────┐
        │  Web 前端    │  │  AI 问答     │  │  文档管理    │
        │  (Next.js)  │  │  (FastAPI)  │  │  (FastAPI)  │
        └────────────┘  └─────┬──────┘  └──────┬─────┘
                              │                │
                    ┌─────────┼────────┐       │
                    ↓         ↓        ↓       ↓
              ┌─────────┐ ┌──────┐ ┌──────┐ ┌──────────┐
              │ Vector  │ │Redis │ │ LLM │ │ PostgreSQL │
              │ Store   │ │Cache │ │ API │ │ + pgvector│
              └─────────┘ └──────┘ └──────┘ └──────────┘
```

### AI 服务特点
- **IO 密集型而非 CPU 密集型** — 主要瓶颈在等待外部 API
- **长连接多** — 流式输出占用长连接
- **突发流量** — 用户行为不可预测
- **成本敏感** — 每次 API 调用都是真金白银

## 4. LLM 缓存策略

### 缓存层级
```
请求 → L1: 精确匹配缓存（Redis, SHA256 key）
      → L2: 语义缓存（向量相似度，相似问题命中）
      → L3: 调用 LLM API
```

### 精确匹配 vs 语义缓存
```python
class TwoLevelCache:
    def __init__(self, redis_client, vector_store):
        self.redis = redis_client
        self.vs = vector_store

    async def get(self, prompt: str) -> dict | None:
        # L1: 精确匹配
        exact_key = f"exact:{hashlib.sha256(prompt.encode()).hexdigest()}"
        cached = await self.redis.get(exact_key)
        if cached:
            return json.loads(cached)

        # L2: 语义相似度匹配
        similar = await self.vs.search(prompt, top_k=1)
        if similar and similar[0]["similarity"] > 0.97:
            return json.loads(similar[0]["response"])

        return None

    async def set(self, prompt: str, response: dict):
        # 同时写入两层缓存
        key = f"exact:{hashlib.sha256(prompt.encode()).hexdigest()}"
        value = json.dumps(response)
        await self.redis.setex(key, 3600, value)
        await self.vs.add(prompt, value)  # 用于语义匹配
```

## 5. 限流与重试

### 指数退避重试
```python
import random

async def call_llm_with_backoff(prompt: str, max_retries: int = 5) -> str:
    """带 jitter 的指数退避"""
    for attempt in range(max_retries):
        try:
            return await llm_client.chat(prompt)
        except RateLimitError:
            if attempt == max_retries - 1:
                raise
            # 指数退避 + 随机抖动，防止 thundering herd
            wait = (2 ** attempt) + random.uniform(0, 1)
            print(f"被限流，等待 {wait:.1f}s 后重试 (第 {attempt+1} 次)")
            await asyncio.sleep(wait)
        except (ConnectionError, TimeoutError) as e:
            if attempt == max_retries - 1:
                raise
            wait = (2 ** attempt) + random.uniform(0, 0.5)
            await asyncio.sleep(wait)
```

### 令牌桶限流
```python
class TokenBucket:
    """令牌桶算法 — 平滑限流"""
    def __init__(self, rate: int, capacity: int):
        self.rate = rate              # tokens/秒
        self.capacity = capacity       # 桶容量
        self.tokens = capacity
        self.last_refill = time.monotonic()

    async def acquire(self) -> bool:
        now = time.monotonic()
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_refill = now

        if self.tokens >= 1:
            self.tokens -= 1
            return True
        return False
```

```python
# 成本追踪

                    ["model"])

# 使用示例
async def traced_llm_call(prompt: str, model: str = "gpt-4"):
    start = time.monotonic()
    try:
        response = await llm_client.chat(prompt)
        tokens_used.labels(model=model, type="input").inc(response.input_tokens)
        tokens_used.labels(model=model, type="output").inc(response.output_tokens)
        llm_latency.labels(model=model, status="success").observe(time.monotonic() - start)
        return response
    except Exception:
        llm_latency.labels(model=model, status="error").observe(time.monotonic() - start)
        raise
```

### Prompt Injection 防护
```python
def sanitize_user_input(text: str) -> str:
    """基础防护 — 检测常见注入模式"""
    suspicious_patterns = [
        r"忽略.*指令",
        r"ignore.*instruction",
        r"system:\s*",
        r"<\|im_start\|>",
        r"<\|im_end\|>",
        r"\[INST\]",
        r"\[/INST\]",
    ]
    for pattern in suspicious_patterns:
        if re.search(pattern, text, re.IGNORECASE):
            logger.warning("检测到疑似 Prompt Injection", text=text[:100])
            # 选项：拒绝请求 / 标记为可疑 / 清理后继续
            raise ValueError("输入包含不被允许的内容")
    return text
```

### 输出审核 (Guardrails)
```python
class OutputGuard:
    """输出内容安全审核"""
    def __init__(self, moderation_client):
        self.moderation = moderation_client

    async def check(self, text: str) -> bool:
        """返回 True 表示安全，False 表示需要拦截"""
        # 1. OpenAI Moderation API
        result = await self.moderation.moderations.create(input=text)
        if result.results[0].flagged:
            return False

        # 2. 自定义敏感词检查
        sensitive_words = ["密码", "密钥", "内网IP"]
        for word in sensitive_words:
            if word in text:
                return False

        return True

    async def safe_response(self, unsafe: bool):
        if unsafe:
            return "抱歉，回答中包含不适合展示的内容，请重新提问。"
```
